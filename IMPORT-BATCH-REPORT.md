  if (data.type === 'Feature' && data.geometry) return true;
  if (data.type === 'FeatureCollection' && Array.isArray(data.features)) return true;
  if (data.type === 'Polygon' && Array.isArray(data.coordinates)) return true;
  
  return false;
};
